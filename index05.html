Sub FetchMovieDetails()
    Dim xhr As Object
    Dim title As String
    Dim url As String
    Dim json As String
    
    ' Initialize XMLHttpRequest object
    Set xhr = CreateObject("MSXML2.ServerXMLHTTP.6.0")
    
    ' Read selected movie ID from cell B3
    title = Sheets("Dashboard").Range("B3").Value
    
    ' Construct the API URL
    url = "http://localhost:3000/movie-details?title=" & title
    
    ' Make a GET request to the API
    xhr.Open "GET", url, False
    xhr.send
    
    ' Get the response JSON as a string
    json = xhr.responseText
    
    ' TODO: Parse and display the JSON in Excel cells
    ' For example purposes, we just display it in a message box
    MsgBox json
    
End Sub
